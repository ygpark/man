CREATE TABLE `test_weblog` (
`idx` BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT,
`remote_host` VARCHAR(50) NULL DEFAULT NULL,
`remote_host_int` BIGINT(20) NOT NULL,
`remote_logname` VARCHAR(50) NULL DEFAULT NULL,
`remote_user` VARCHAR(50) NULL DEFAULT NULL,
`timestamp` TIMESTAMP NULL DEFAULT NULL,
`method` VARCHAR(50) NULL DEFAULT NULL,
`url` TEXT NULL,
`http_ver` VARCHAR(50) NULL DEFAULT NULL,
`status` VARCHAR(50) NULL DEFAULT NULL,
`bytes_tx` VARCHAR(50) NULL DEFAULT NULL,
`referer` TEXT NULL,
`user_agent` VARCHAR(2048) NULL DEFAULT NULL,
`logfile` VARCHAR(1024) NULL DEFAULT NULL,
PRIMARY KEY (`idx`),
INDEX `remote_host_int` (`remote_host_int`)
)
COLLATE='utf8_general_ci'
ENGINE=MyISAM;
